193 class TestConvergence {
194 public:
195  double old_stress;
196  TestConvergence(const double tol = 1e-4, const unsigned maxiterations = 100)
197  : tolerance(tol),
198  maxiterations(maxiterations)
199  { reset(); }
200  virtual ~TestConvergence() {}
201 
202 public:
203  virtual bool operator()(const double new_stress, std::valarray<double> & X, std::valarray<double> & Y)
204  {
205  COLA_UNUSED(X);
206  COLA_UNUSED(Y);
207 
208  iterations++;
209  //std::cout<<"iteration="<<iterations<<", old_stress="<<old_stress<<", new_stress="<<new_stress<<std::endl;
210  if (old_stress == DBL_MAX) {
211  old_stress = new_stress;
212  return iterations >= maxiterations;
213  }
214  // converged if relative decrease in stress falls below threshold
215  // or if stress increases (shouldn't happen for straight majorization)
216  bool converged =
217  (old_stress - new_stress) / (new_stress + 1e-10) < tolerance
218  || iterations > maxiterations;
219  old_stress = new_stress;
220  return converged;
221  }
222  void reset() {
223  old_stress = DBL_MAX;
224  iterations = 0;
225  }
226  const double tolerance;
227  const unsigned maxiterations;
228  unsigned iterations;
229 };
