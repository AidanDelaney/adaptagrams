158  inline vpsc::Rectangle *getMinEdgeRect(const vpsc::Dim dim)
159  {
160  if (minEdgeRect[dim])
161  {
162  delete minEdgeRect[dim];
163  }
164  minEdgeRect[dim] = new vpsc::Rectangle(bounds);
165 
166  // Set the Min and Max positions to be the min minus an offset.
167  double edgePosition = minEdgeRect[dim]->getMinD(dim);
168  minEdgeRect[dim]->setMinD(dim, edgePosition - rectBuffer);
169  minEdgeRect[dim]->setMaxD(dim, edgePosition);
170 
171  return minEdgeRect[dim];
172  }
173 
174  inline vpsc::Rectangle *getMaxEdgeRect(const vpsc::Dim dim)
175  {
176  if (maxEdgeRect[dim])
177  {
178  delete maxEdgeRect[dim];
179  }
180  maxEdgeRect[dim] = new vpsc::Rectangle(bounds);
181 
182  // Set the Min and Max positions to be the max plus an offset.
183  double edgePosition = maxEdgeRect[dim]->getMaxD(dim);
184  maxEdgeRect[dim]->setMaxD(dim, edgePosition + rectBuffer);
185  maxEdgeRect[dim]->setMinD(dim, edgePosition);
186 
187  return maxEdgeRect[dim];
188  }
